create table vehiculos(
	id serial primary key,
	patente varchar not null unique,
	marca varchar not null,
	activo boolean default true
);

/*
Crear una API REST de gestión de vehículos. Se deben desarrollar las siguientes rutas.
- GET /vehiculo -> Listar Vehículos.
- POST /vehiculo -> Registrar Vehículos.
- PUT /vehiculo/:id -> Modificar Vehículos.
- DELETE /vehiculo/:id -> Eliminar Vehiculos.
- POST /vehiculo/:id/activar -> Ruta activar debe cambiar el campo activo a true
- POST /vehiculo/:id/desactivar -> Ruta desactivar debe cambiar el campo activo a false
 */
